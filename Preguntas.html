groups = [df[df['Heating'] == heating]['SalePrice'] for heating in df['Heating'].unique()]
f_statistic, p_value = f_oneway(*groups)

print(f"Estadstico F: {f_statistic:.2f}")
print(f"Valor p: {p_value:.4f}")

if p_value < 0.05:
    print("Existe una diferencia significativa en los precios entre los diferentes tipos de calefaccin.")
else:
    print("No existe una diferencia significativa en los precios entre los diferentes tipos de calefaccin.")
